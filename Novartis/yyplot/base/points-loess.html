### Make Y~Y Plot
yyplot(
  x.column = "Petal.Length",
  y.column = "Sepal.Length",
  split.column = "Species",
  data=iris,
  title.of.all.plots = "Iris Data Set",
  xlab="Petal Length",
  ylab="Sepal Length",
  loess.list = list(
    color = "blue",
    span = c(5/6),
    legend = "Loess Line"
  )
)
#dev.off()
